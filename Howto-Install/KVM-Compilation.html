Download qemu-kvm source from qemu-kvm-*.tar.gz. 

$ sudo apt-get install libpci3 pciutils-dev
$ sudo apt-get install libncurses5-dev 
$ sudo apt-get install libvdeplug2 libvdeplug2-dev 
$ sudo apt-get install gcc libsdl1.2-dev zlib1g-dev libasound2-dev linux-kernel-headers pkg-config libgnutls-dev libpci-dev
## glib-2.0 required to compile QEMU
$ sudo apt-get install libgtk2.0-dev 
$ cd /src3/kernel/
$ tar zxvf qemu-kvm-*.tar.gz 
$ cd qemu-kvm-*
$ mkdir objects; cd objects
## $ ../configure --help
$ emacs myconf&
$ cat myconf
#! /bin/bash 
#../configure --prefix=/usr/local --source-path=.. --target-list=x86_64-linux-user,x86_64-softmmu --enable-kvm --enable-linux-user --enable-vde --enable-vhost-net --enable-docs
$ chmod 755 myconf 
$ myconf 
$ make 
$ echo $? 
# If return 0
$ sudo make install 
$ cd x86_64-softmmu
$ qemu-system-x86_64 -hda /src3/KVM/Resize/Debian-Mini.img
$ sudo make install 
$ cd /usr/local/bin 
$ sudo ln qemu-system-x86_64 kvm 
$ cd /src3/kernel/qemu-kvm-*/objects 
$ make clean
